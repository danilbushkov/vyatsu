% Изменение стилей списков

\usepackage{enumitem}



\newcommand{\labelv}{--}
\newcommand{\labels}{0.5em}


\setlist[itemize]{
	itemsep=0pt,
	parsep=0pt,
	topsep=0pt,
	labelsep=\labels,
	label=\labelv,
	itemindent=\parindent+\labels+\labelwidth,
	leftmargin=0pt,
	align=left,
}


\setlist[enumerate,1]{
	itemsep=0pt,
	parsep=0pt,
	topsep=0pt,
	labelsep=\labels,
	label=\arabic*),
	ref=\arabic*,
	leftmargin=\parindent+\labelwidth+\labelsep,
	align=left,
}

\setlist[enumerate,2]{
	itemsep=0pt,
	parsep=0pt,
	topsep=0pt,
	labelsep=\labels*2,
	label=\theenumi.\arabic*),
	leftmargin=\labelwidth+\labelsep,
	align=left,
}
